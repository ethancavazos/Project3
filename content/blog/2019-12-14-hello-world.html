#Find all 3-mers in these two sequences

my_seq1 = "ATCATCATG"
my_seq2 = "CAGCCCAATCAGGCTCTACTGCCACTAAACTTACGCAGGATATATTTACGCCGACGTACT"
def kmers(seq, k):

    counts = {}
    kmers = len(seq) - k + 1

    for i in range(kmers):

        kmer = seq[i:i+k]

        if kmer not in counts:
            counts[kmer] = 0
        counts[kmer] += 1
